
void mainImage(out vec4 fragColor, in vec2 fragCoord )  {
    vec4 f = vec4(0,0,0,0);
    float t = iTime;
    
    for (float i=0.; i < 20.; i+=.1) {
        if (fract(sin(vec4(6,9,1,5.*t)*i) * 9e2).w <= .98){
            f +=   fract(sin(vec4(6,9,1,0)*i) * 9e2)/8e2/ pow(length(fragCoord/iResolution.y - 3.*fract( fract(sin(vec4(6,9,1,5)*i) * 9e2).zw )), 1.2 ); 
        }
    }
    fragColor = vec4(f.xyz / 2., 1.0);
}
